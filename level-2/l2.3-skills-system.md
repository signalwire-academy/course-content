## Complete Solution

```python
#!/usr/bin/env python3
"""Lab 2.3: Skills system."""

from signalwire_agents import AgentBase


class SkillsAgent(AgentBase):
    def __init__(self):
        super().__init__(name="skills-agent")

        self.prompt_add_section(
            "Role",
            "You are a helpful assistant with access to various skills. "
            "Use your skills to help users with time, math, and web information."
        )

        self.prompt_add_section(
            "Capabilities",
            bullets=[
                "Tell the current date and time",
                "Perform mathematical calculations",
                "Search the web for information"
            ]
        )

        self.add_language("English", "en-US", "rime.spore")

        # Add skills
        self.add_skill(
            "datetime",
            {
                "timezone": "America/New_York",
                "format": "12-hour"
            }
        )

        self.add_skill("math")

        self.add_skill(
            "web_search",
            {
                "max_results": 3
            }
        )


if __name__ == "__main__":
    agent = SkillsAgent()
    agent.run()
```
